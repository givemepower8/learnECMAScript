named function

```js
function say(message) {
  console.log(message);
}
say('hello world');
```

or we create a function instance

```js
// in ECMA Script 6, we use const instead of var
const say = function(message) {
  console.log(message);
};

say('hello world');
```

```js
const say = (message) => {
  console.log(message);
};

say('hello world');
```

function often returns something

```js
const max = function(a, b) {
  console.log(arguments);
  if (a > b) {
    return a;
  }

  return b;
};

console.log(max(1));
console.log(max(1, 3));
console.log(max(4, 2));
console.log(max(2, 7, 12));
console.log(max('((',"'", 7 ));
```
